73void cleanFilesByName(const fs::path &directoryPath)
74{
75 // Clear any leftover newline from prior input and prompt the user
76 cin.ignore(numeric_limits<streamsize>::max(), '\n');
77 cout << BOLD << WHITE << "Enter a name to search for in filenames (press Enter to auto-detect common names):\n>>: " << RESET;
78 string name;
79 getline(cin, name);
80
81 vector<fs::path> matches; // Files matching the name
82 size_t moved = 0; // Count of moved files
83 size_t skipped = 0; // Count of skipped files
84 vector<fs::path> skippedFiles; // List of skipped files
85
86 // Branch 1: explicit user-supplied name search
87 if (!name.empty())
88 {
89 string needle = toLower(name);
90 for (const auto &entry : fs::directory_iterator(directoryPath))
91 {
92 if (!entry.is_regular_file())
93 continue;
94 string fname = entry.path().filename().string();
95 if (toLower(fname).find(needle) != string::npos)
96 {
97 matches.push_back(entry.path());
98 }
99 }
100
101 if (matches.empty())
102 {
103 cout << YELLOW << "No files found containing '" << name << "'.\n" << RESET;
104 cout << YELLOW << "Press Enter to return to the menu..." << RESET;
105 cin.get();
106 return;
107 }
108
109 // Create destination directory named after provided name (sanitize)
110 string dirName = name;
111 for (auto &c : dirName)
112 if (c == '/' || c == '\\')
113 c = '_'; // replace slashes with underscores
114
115 fs::path destDir = directoryPath / dirName;
116 error_code ec;
117 if (!fs::exists(destDir))
118 fs::create_directory(destDir, ec);
119 if (ec)
120 {
121 cerr << RED << "Failed to create directory '" << destDir << "': " << ec.message() << RESET << "\n";
122 return;
123 }
124
125 // Move matched files
126 for (const auto &src : matches)
127 {
128 fs::path dest = destDir / src.filename();
129 if (fs::exists(dest))
130 {
131 ++skipped;
132 skippedFiles.push_back(src.filename());
133 continue;
134 }
135 error_code moveEc;
136 fs::rename(src, dest, moveEc);
137 if (moveEc)
138 {
139 ++skipped;
140 skippedFiles.push_back(src.filename());
141 }
142 else
143 {
144 ++moved;
145 }
146 }
147 }
148 else
149 {
150 // Auto-detect common name tokens
151 map<string, int> tokenCount;// Token frequency map
152
153 // Build ignore set from dynamic tokens
154 const auto& ignoreTokensVec = getIgnoreTokens(); // from include/ignoreTokens.hpp
155 unordered_set<string> ignoreSet(ignoreTokensVec.begin(), ignoreTokensVec.end());
156
157 // Tokenize stems and count tokens of length >= 4
158 for (const auto &entry : fs::directory_iterator(directoryPath))
159 { //loop through files
160 if (!entry.is_regular_file()){// Check for regular file
161 cout<< DIM << "Skipping non-regular file: " << entry.path().filename().string() << RESET << "\n";
162 continue;
163 }
164 string stem = entry.path().stem().string();
165 string lower = toLower(stem);// Lowercase stem
166
167 // Split on non-alphanumeric
168 string token;
169 for (char ch : lower)
170 {
171 if (isalnum(static_cast<unsigned char>(ch)))// Alphanumeric check
172 {
173 token.push_back(ch);
174 }
175 else
176 {
177 if (token.size() >= 4 && ignoreSet.find(token) == ignoreSet.end()){
178 tokenCount[token]++;
179 token.clear();
180 }
181 }
182 }
183
184 // Handle final token
185 if (token.size() >= 4 && ignoreSet.find(token) == ignoreSet.end()){
186 tokenCount[token]++;
187 }
188 // Also count whole stem if long enough
189 if (lower.size() >= 4 && ignoreSet.find(lower) == ignoreSet.end()){
190 tokenCount[lower]++;
191 }
192 }
193
194 // Collect tokens that appear multiple times
195 vector<pair<string, int>> common;
196 for (auto &kv : tokenCount)
197 if (kv.second >= 2)
198 common.emplace_back(kv.first, kv.second);
199
200 // Sort by frequency descending
201 sort(common.begin(), common.end(), [](auto &a, auto &b) {
202 return a.second > b.second;
203 });
204
205 if (common.empty())
206 {
207 cout << YELLOW << "No common name tokens detected. Nothing to move.\n" << RESET;
208 cout << YELLOW << "Press Enter to return to the menu..." << RESET;
209 cin.get();
210 return;
211 }
212
213 // Limit to top 10 tokens to avoid over-creating folders
214 size_t limit = min<size_t>(10, common.size());
215 for (size_t i = 0; i < limit; ++i)
216 {
217 const string token = common[i].first;
218
219 // Find files that contain this token
220 vector<fs::path> found;
221 for (const auto &entry : fs::directory_iterator(directoryPath))
222 {
223 if (!entry.is_regular_file())
224 continue;
225 string fnameLower = toLower(entry.path().filename().string());
226 if (fnameLower.find(token) != string::npos && ignoreSet.find(token) == ignoreSet.end())
227 found.push_back(entry.path());// Add to found list
228 }
229
230 if (found.size() < 2)
231 continue; // Skip tokens that don't represent groups
232
233 // Make directory named after token
234 fs::path destDir = directoryPath / token;
235 error_code ec;
236 if (!fs::exists(destDir))
237 fs::create_directory(destDir, ec);
238 if (ec)
239 {
240 cerr << RED << "Failed to create directory '" << destDir << "': " << ec.message() << RESET << "\n";
241 continue;
242 }
243
244 // Move files for this token
245 for (const auto &src : found)
246 {
247 fs::path dest = destDir / src.filename();
248 if (fs::exists(dest))
249 {
250 ++skipped;
251 cout<< DIM << "Skipping file due to name conflict: " << src.filename().string() << RESET << "\n";
252 skippedFiles.push_back(src.filename());
253 continue;
254 }
255 error_code moveEc;
256 fs::rename(src, dest, moveEc);
257 if (moveEc)
258 {
259 ++skipped;
260 cout<< DIM << "Skipping file due to error: " << src.filename().string() << RESET << "\n";
261 skippedFiles.push_back(src.filename());
262 }
263 else
264 {
265 ++moved;
266 }
267 }
268 }
269 }
270
271 // Print results
272 cout << GREEN << "Moved: " << moved << RESET << " " << YELLOW << "Skipped: " << skipped << RESET << "\n";
273 if (!skippedFiles.empty())
274 {
275 cout << DIM << "Skipped files (name conflicts or errors):\n" << RESET;
276 for (const auto &s : skippedFiles)
277 cout << " - " << s.string() << "\n";
278 }
279
280 cout << YELLOW << "Press Enter to return to the menu..." << RESET;
281 cin.get();
282
283 return;
284}
